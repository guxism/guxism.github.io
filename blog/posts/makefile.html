Makefile 手册

Make 可以用来管理 C/C++ 编译 之外的其他事务, 比如以下例子, 检查版本, 登录到主机编译, 下载二进制. 这种通常用脚本做的事完全可以搬到 makefile.

version = $(shell cat VERSION.txt)
bin: VERSION.txt
	ssh ${REMOTE_USER}@${REMOTE_IP} "cd ${REMOTE_DIR} && git checkout tags/${version} && make release"
	wget ${REMOTE_IP}/release -O release
	tar -xzvf -C tmp
	mv tmp/release/bin .
	rm tmp -rf

Makefile 提供文件依赖检查机制, 但这不是最关键的优点. 最关键的优点是它语法相对简单(对比 CMake, Gulp), 符合直觉, 方便管理, 可以当备忘录使用(目标就是每一条备忘录的标签).

本文列举了不常见但实用的语法.

把同一目录下的 *.s 编译成 *(批量操作)

asm_src := $(wildcard *.s)
bins := ${asm_src:%.s=%}
$(info ${bins})
$(info ${asm_src})

all: ${bins}

${bins}: %: %.s
	gcc $^ -no-pie -o $@

交叉编译(条件变量)

ifeq (${ARCH},arm64)
OUT_DIR := ${ARCH}/
CROSS_COMPILE := aarch64-linux-gnu-
CXX := ${CROSS_COMPILE}g++
CC := ${CROSS_COMPILE}gcc
endif

run:

make ARCH=arm64

.PHONY 目标

.PHONY: config clean run
clean:
	@rm out -rf

抽象(自定义函数)

define MKDIR?
	$(shell mkdir -p $1)
endef
$(call MKDIR?, ${OUT_DIR})

内置字符串处理函数

run: out/$(basename ${x})
	@make --silent  && cd $(dir $<) && ./$(notdir $<)

run:

make run x=test/test_addr.cpp

搭配 awk

docs/%.c.html: .tmp/docs/%.c
	pycco $< > .tmp/$@
	$(eval a=$(shell awk '{print $$4}' .tmp/$@))
	echo $a

或

docs/%.c.html: .tmp/docs/%.c
	pycco $< > .tmp/$@
	a=$(shell awk '{print $$4}' .tmp/$@);echo $$a
  • eval 将 a 设置为 makefile 变量.
  • 没有 eval, a 就是 bash 环境变量, 每一行的环境都不一样, 如果将 echo $$a 放在下一行, a 就是未定义.
  • $a 是 makefile 变量, $$a 是 bash 变量

让中间文件自动消失

.tmp/%.i:
	mkdir -p .tmp
	touch $@
%.txt: .tmp/%.i
	cp $< $@

运行:

➜  make a.txt
mkdir -p .tmp
touch .tmp/a.i
cp .tmp/a.i a.txt
rm .tmp/a.i

调用子目录

可能我这解决得不是特别好：

targets := all clean
subdirs := $(wildcard */.)

prog := a.out

all_local: ${prog}

clean_local:
	rm ${prog}

${targets}: ${subdirs}
	${MAKE} ${MAKECMDGOALS}_local

${subdirs}:
	${MAKE} -C $@ $(MAKECMDGOALS)

.PHONY: all ${subdirs} all_local clean_local

CFLAGS += -fuse-ld=bfd

${prog}: main.c sub/libsub.a add/libadd.a
	gcc -o $@ $^ ${CFLAGS}
