239. 滑动窗口最大值

给你一个整数数组 nums, 有一个大小为 k 的滑动窗口从数组的最左侧移动到数组的最右侧. 你只可以看到在滑动窗口内的 k 个数字. 滑动窗口每次只向右移动一位.

返回滑动窗口中的最大值.

示例 1：

输入：nums = [1,3,-1,-3,5,3,6,7], k = 3
输出：[3,3,5,5,6,7]
解释：
滑动窗口的位置                最大值
---------------               -----
[1  3  -1] -3  5  3  6  7       3
 1 [3  -1  -3] 5  3  6  7       3
 1  3 [-1  -3  5] 3  6  7       5
 1  3  -1 [-3  5  3] 6  7       5
 1  3  -1  -3 [5  3  6] 7       6
 1  3  -1  -3  5 [3  6  7]      7

struct Tracer
{
    Tracer() : empty(true) {}
    int max;
    bool empty;
    multiset<int> set;
    void push(int x)
    {
        if (empty)
        {
            max = x;
        }
        else if (x > max)
        {
            set.insert(max);
            max = x;
        }
        else
        {
            set.insert(x);
        }
        empty = false;
    }
    void pop(int x)
    {
        if (x == max)
        {
            if (!set.empty())
            {
                auto i = set.end();
                --i;
                max = *i;
                set.erase(i);
            }
            else
            {
                empty = true;
            }
        }
        else if (!set.empty())
        {
            auto i = set.find(x);
            set.erase(i);
        }
    }
};

public:
vector<int> maxSlidingWindow(vector<int> &nums, int k)
{
    if (nums.empty() || k <= 0)
        return {};
    auto front = nums.begin();
    auto back = nums.begin();
    Tracer tracer;
    vector<int> res;
    tracer.push(*front++);
    while (front - back < k)
    {
        tracer.push(*front);
        ++front;
    }
    res.push_back(tracer.max);
    while (front < nums.end())
    {
        tracer.pop(*back);
        ++back;
        tracer.push(*front);
        ++front;
        res.push_back(tracer.max);
    }
    return res;
}

看哪里还能优化, 首先看这个问题的规律, 第一个问题是： (4) 的 multiset 是否能用更高效的东西来替换？

进入考智商环节, 这些规律如下：

  1. tracer 的每个元素都有一个生命周期(lifetime)
  2. lifetime(A[i+1]) > lifetime(A[i])
  3. 如果 A[i+1] 是新元素且大于 tracer.max, 则可以清空 tracer.max. 因为 A[i+1] 比 tracer 中所有元素 S 都活得久, S < A[i+1] 所以 S 没必要存在
  4. 如果 A[i+1] < A[i], 由于 2, A[i] 可能先被弹出, 到时候第二大值必须放在 max 位, 这个值是比 A[i] 更新的元素. 因此所有比 A[i] 新且小于 A[i] 的元素, 都放在 A[i] 后面.
  5. 因为, 根据 3, 更老的元素在 A[i] 被确立为最大值的时候被清空.
    • 如果不清空, 会怎样？
    • 答：不清空会导致一个已经不在窗口上的元素, 在 tracer.max 被弹出之后, 可能会被选为最大值.
    • 没人规定我不能清空, 因为清空不影响 tracer 的作用, 所以清空可以减低维护难度
  6. 在 A[i] 被弹出之后, 第 2 大值 X 至少有 1 个迭代的存活时间. 那么在 X 后面的 Y 至少有 2 个迭代存活时间.
  7. 因为第二大值可能是 A[i+1].
  8. 继续 4, 比 A[i] 新且小的元素 X, 放在 A[i] 后面, 那么 X 可以放在第一个比它大的数字后面.
  9. 因为, 对于 X a b c d Y A[i], 如果 a b c d 都比 X 小, 因为 X 生命周期更长, 所以 a b c d 没有存在的必要.
  10. 二, 因为 Y 可能是下一个最大值, 所以不能删, 根据 5, A[i] 被弹出之前, Y 都是有存在必要的.

struct Tracer
{
    Tracer(int k) : buf(new int[k]), head(0), cap(k), bufsize(0) {}
    ~Tracer() { delete[] buf; }
    int *buf;
    int head;
    int cap;
    int bufsize;
    inline int top() { return buf[head]; }
    void push(int x)
    {
        if (!bufsize || x > buf[head])
        {
            buf[head] = x;
            bufsize = 1;
        }
        else
        {
            int tail = head + bufsize - 1 + cap;
            while (tail % cap != head && buf[tail % cap] < x)
            {
                --bufsize;
                --tail;
            }
            buf[(tail + 1) % cap] = x;
            ++bufsize;
        }
    }
    void pop()
    {
        if (bufsize)
        {
            head = (head + 1) % cap;
        }
        --bufsize;
    }
};
vector<int> maxSlidingWindow(vector<int> &nums, int k)
{
    if (nums.empty() || k <= 0 || nums.size() < (size_t)k)
        return {};

    auto front = nums.begin();
    auto back = nums.begin();
    Tracer t(k);
    vector<int> res;
    t.push(*front++);
    while (front - back < k)
    {
        t.push(*front);
        ++front;
    }
    res.push_back(t.top());
    while (front < nums.end())
    {
        if (t.top() == *back)
        {
            t.pop();
        }
        ++back;

        t.push(*front);
        ++front;
        res.push_back(t.top());
    }
    return res;
}
