面试后做一些改进

int compare(const string &v1, const string &v2)
{
    auto vec1 = vectorize(v1);
    auto vec2 = vectorize(v2);
#

制造模式: xxxx.

    vec1.push_back('.');
    vec2.push_back('.');

    auto p1 = vec1.begin();
    auto p2 = vec2.begin();
#

越过第一个点 “.”

    p1++;
    p2++;

    while (p1 != vec1.end())
    {
        auto e1 = p1;
        auto e2 = p2;
#

探索每一段 xxx. 的终点

        while (*e1 != '.')
            ++e1;
        while (*e2 != '.')
            ++e2;

        auto len1 = e1 - p1;
        auto len2 = e2 - p2;
#

如果长度不相等, 返回最长的

        if (len1 != len2)
        {
            if (len1 > len2)
                return 1;
            else if (len1 < len2)
                return -1;
        }
#

否则, 逐一比较

        while (p1 <= e1)
        {
            if (*p1 > *p2)
            {
                return 1;
            }
            else if (*p1 < *p2)
            {
                return -1;
            }
            ++p1;
            ++p2;
        }

    }

    return 0;
}
