#

select()

int select(int nfds, fd_set *readfds, fd_set *writefds,
               fd_set *exceptfds, struct timeval *timeout);

man select:

  • select() 是指一条线程监听多个文件描述符的 readiness notification 接口, ready 是指对文件描述符读写无需阻塞, 也不返回 EAGAIN 或 EWOULDBLOCK,
  • 监听三组文件描述符
  • 文件描述符组会被原地修改, 因此每次调用之前应该重置文件描述符组(set)
  • 第一个参数应该设置为文件描述符的最大数+1, 有三组, 但只有一个 nfds
  • select() 阻塞到
    • 有一个文件描述符处于就绪状态
    • 被信号打断
    • 超时
  • select() 有描述符数量限制 FD_SETSIZE, poll() 没有此限制
  • 如果使用 select() 同时监听 signal 和 I/O, 则可能形成竞争关系, 这个问题将在附录展开说明.

int set_nonblocking(int fd)
{
    int flags;
    if (-1 == (flags = fcntl(fd, F_GETFL, 0)))
        flags = 0;
    return fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

int create_socket(int port)
{
    int listener_sockfd;
    struct sockaddr_in addr;

    listener_sockfd = socket(AF_INET, SOCK_STREAM, 0);

    if (listener_sockfd < 0)
    {
        return -1;
    }

    addr.sin_family = AF_INET;
    addr.sin_port = htons(port);
    addr.sin_addr.s_addr = inet_addr("0.0.0.0");

    int optval = 1;
    setsockopt(listener_sockfd, SOL_SOCKET, SO_REUSEPORT, &optval, sizeof(optval));

    if (set_nonblocking(listener_sockfd) == -1)
    {
        return -1;
    }

    if (bind(listener_sockfd, (struct sockaddr *)&addr, sizeof(addr)))
    {
        return -1;
    }

    if (listen(listener_sockfd, 5))
    {
        return -1;
    }

    return listener_sockfd;
}

附录: 信号函数造成的竞争关系

void handler(int unused) { (void)unused; }

int main_thread()
{
    fd_set rfds;
    struct timeval tv;
    int retval;
    char buf[128];
    FD_ZERO(&rfds);

    tv.tv_sec = 5;
    tv.tv_usec = 0;

    struct sigaction sa;
    sa.sa_handler = handler;
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = 0;
    sigaction(SIGINT, &sa, NULL);

    FD_SET(0, &rfds);
#

这里会出现一个问题, 如果中断在 select() 之前发生, 之后 1 分钟内用户不再按 Ctrl + C, 则下面的 errno 就不是 EINTR, 可以说信号处理函数与主线程形成了竞争关系; 正确做法是阻塞信号, select 之后解开阻塞; pselect() 提供了这种功能, 相当于

sigset_t sigsaved;

sigprocmask(SIG_SETMASK, &sigmask, &sigsaved);
ready = select(nfds, &readfds, &writefds, &exceptfds, timeout);
sigprocmask(SIG_SETMASK, &sigsaved, NULL);

    retval = select(1, &rfds, NULL, NULL, &tv);
    if (retval == -1)
    {
        if (errno == EINTR)
            printf("select() was interrupted\n");
        else
            perror("select()");
    }
    else if (retval)
    {
        bzero(buf, 128);
        read(0, buf, 128);
        printf("You say: %s\n", buf);
    }
}
